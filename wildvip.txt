-- Safe Drawing wrapper
local function SafeDrawing(type, props)
    if not DrawingAvailable then return nil end
    local success, obj = pcall(function()
        local d = Drawing.new(type)
        if props then
            for k, v in pairs(props) do
                d[k] = v
            end
        end
        return d
    end)
    return success and obj or nil
end

-- Get closest player to crosshair
local function GetClosestPlayer()
    local closest = nil
    local shortest = Features.AimFOV
    local mousePos = UserInputService:GetMouseLocation()
    
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= player then
            local success, result = pcall(function()
                if not p.Character then return nil end
                local hum = p.Character:FindFirstChildOfClass("Humanoid")
                if not hum or hum.Health <= 0 then return nil end
                
                local targetPart
                if Features.AutoHeadshot then
                    targetPart = p.Character:FindFirstChild("Head") or p.Character:FindFirstChild("HumanoidRootPart")
                else
                    targetPart = p.Character:FindFirstChild("HumanoidRootPart") or p.Character:FindFirstChild("Head")
                end
                
                if not targetPart then return nil end
                
                -- Wall check
                if not Features.Wallbang then
                    local ray = Ray.new(camera.CFrame.Position, (targetPart.Position - camera.CFrame.Position).Unit * 1000)
                    local hit = workspace:FindPartOnRayWithIgnoreList(ray, {player.Character})
                    if hit and not hit:IsDescendantOf(p.Character) then
                        return nil
                    end
                end
                
                local pos, onScreen = camera:WorldToViewportPoint(targetPart.Position)
                if not onScreen then return nil end
                
                local dist = (Vector2.new(pos.X, pos.Y) - mousePos).Magnitude
                if dist < shortest then
                    return {Player = p, Part = targetPart, Distance = dist, ScreenPos = pos}
                end
                return nil
            end)
            
            if success and result then
                closest = result
                shortest = result.Distance
            end
        end
    end
    
    return closest
end

-- Silent Aim (modifies bullet direction without moving camera)
local function DoSilentAim()
    if not Features.SilentAim then return end
    
    local target = GetClosestPlayer()
    if not target then return end
    
    -- This would hook into the game's shooting mechanics
    -- For now, we'll use a simplified approach
    return target.Part.Position
end

-- --- MAIN LOOP ---
RunService.RenderStepped:Connect(function()
    pcall(function()
        -- Update FOV Circle
        if FOVCircle then
            FOVCircle.Visible = Features.Aimbot or Features.SilentAim
            FOVCircle.Radius = Features.AimFOV
            FOVCircle.Position = UserInputService:GetMouseLocation()
            FOVCircle.Color = Features.ESPColor
        elseif FOVFrame then
            FOVFrame.Visible = Features.Aimbot or Features.SilentAim
            FOVFrame.Size = UDim2.new(0, Features.AimFOV * 2, 0, Features.AimFOV * 2)
            FOVFrame.Position = UDim2.new(0, UserInputService:GetMouseLocation().X - Features.AimFOV, 0, UserInputService:GetMouseLocation().Y - Features.AimFOV)
        end
        
        -- Aimbot
        if Features.Aimbot and UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then
            local target = GetClosestPlayer()
            if target and target.Part then
                local targetPos = target.Part.Position
                local currentCF = camera.CFrame
                local targetCF = CFrame.new(currentCF.Position, targetPos)
                camera.CFrame = currentCF:Lerp(targetCF, 1 / Features.AimSmoothness)
            end
        end
        
        -- Trigger Bot
        if Features.TriggerBot then
            local target = GetClosestPlayer()
            if target and target.Distance < Features.AimFOV * 0.8 then
                -- Auto fire when target in crosshair
                local tool = player.Character and player.Character:FindFirstChildOfClass("Tool")
                if tool then
                    tool:Activate()
                end
            end
        end
        
        -- ESP Update
        if Features.ESP then
            for plr, objects in pairs(ESPObjects) do
                local success, visible, pos, size, health, maxHealth = pcall(function()
                    if not plr.Character then return false end
                    local hum = plr.Character:FindFirstChildOfClass("Humanoid")
                    if not hum or hum.Health <= 0 then return false end
                    
                    local hrp = plr.Character:FindFirstChild("HumanoidRootPart")
                    if not hrp then return false end
                    
                    local screenPos, onScreen = camera:WorldToViewportPoint(hrp.Position)
                    if not onScreen then return false end
                    
                    local head = plr.Character:FindFirstChild("Head")
                    local headPos = head and camera:WorldToViewportPoint(head.Position) or screenPos
                    
                    local sizeY = math.abs(screenPos.Y - headPos.Y) * 2.5
                    local sizeX = sizeY * 0.6
                    
                    return true, screenPos, {X = sizeX, Y = sizeY}, hum.Health, hum.MaxHealth
                end)
                
                if success and visible then
                    -- Box
                    if Features.ESPBoxes then
                        objects.Box.Size = UDim2.new(0, size.X, 0, size.Y)
                        objects.Box.Position = UDim2.new(0, pos.X - size.X/2, 0, pos.Y - size.Y/2)
                        objects.Box.Visible = true
                        objects.Box:FindFirstChildOfClass("UIStroke").Color = Features.ESPColor
                    else
                        objects.Box.Visible = false
                    end
                    
                    -- Tracer
                    if Features.ESPTracers then
                        objects.Tracer.Size = UDim2.new(0, 2, 0, camera.ViewportSize.Y - pos.Y)
                        objects.Tracer.Position = UDim2.new(0, pos.X, 0, pos.Y)
                        objects.Tracer.BackgroundColor3 = Features.ESPColor
                        objects.Tracer.Visible = true
                    else
                        objects.Tracer.Visible = false
                    end
                    
                    -- Name
                    if Features.ESPNames then
                        objects.Name.Text = plr.Name
                        objects.Name.Position = UDim2.new(0, pos.X, 0, pos.Y - size.Y/2 - 15)
                        objects.Name.Visible = true
                    else
                        objects.Name.Visible = false
                    end
                    
                    -- Health
                    if Features.ESPHealth then
                        objects.Health.Text = math.floor(health) .. " HP"
                        objects.Health.Position = UDim2.new(0, pos.X, 0, pos.Y + size.Y/2 + 5)
                        objects.Health.Visible = true
                    else
                        objects.Health.Visible = false
                    end
                    
                    -- Distance
                    if Features.ESPDistance then
                        local dist = player:DistanceFromCharacter(Vector3.new(pos.X, pos.Y, 0))
                        objects.Distance.Text = math.floor(dist) .. "m"
                        objects.Distance.Position = UDim2.new(0, pos.X, 0, pos.Y + size.Y/2 + 20)
                        objects.Distance.Visible = true
                    else
                        objects.Distance.Visible = false
                    end
                else
                    for _, obj in pairs(objects) do
                        obj.Visible = false
                    end
                end
            end
        else
            for _, objects in pairs(ESPObjects) do
                for _, obj in pairs(objects) do
                    obj.Visible = false
                end
            end
        end
        
        -- Speed Hack
        if Features.SpeedHack then
            local hum = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
            if hum then
                hum.WalkSpeed = Features.SpeedValue
            end
        end
        
        -- Fly
        if Features.Fly then
            local char = player.Character
            if char and char:FindFirstChild("HumanoidRootPart") then
                local hrp = char.HumanoidRootPart
                local hum = char:FindFirstChildOfClass("Humanoid")
                if hum then hum.PlatformStand = true end
                
                local moveDir = Vector3.new(0, 0, 0)
                if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir = moveDir + camera.CFrame.LookVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir = moveDir - camera.CFrame.LookVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir = moveDir - camera.CFrame.RightVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir = moveDir + camera.CFrame.RightVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDir = moveDir + Vector3.new(0, 1, 0) end
                if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then moveDir = moveDir - Vector3.new(0, 1, 0) end
                
                if moveDir.Magnitude > 0 then
                    hrp.Velocity = Vector3.new(0, 0, 0)
                    hrp.CFrame = hrp.CFrame + (moveDir.Unit * (Features.FlySpeed * 0.016))
                else
                    hrp.Velocity = Vector3.new(0, 0, 0)
                end
            end
        else
            local hum = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
            if hum and hum.PlatformStand then
                hum.PlatformStand = false
            end
        end
        
        -- No Clip
        if Features.NoClip then
            local char = player.Character
            if char then
                for _, part in pairs(char:GetDescendants()) do
                    if part:IsA("BasePart") then
                        part.CanCollide = false
                    end
                end
            end
        end
        
        -- Fullbright
        if Features.Fullbright then
            Lighting.Brightness = 10
            Lighting.GlobalShadows = false
            Lighting.Ambient = Color3.fromRGB(255, 255, 255)
            Lighting.OutdoorAmbient = Color3.fromRGB(255, 255, 255)
        end
        
        -- FOV Changer
        if Features.FOVChanger then
            camera.FieldOfView = Features.CustomFOV
        end
        
        -- No Flash/Smoke
        if Features.NoFlash then
            for _, effect in pairs(Lighting:GetChildren()) do
                if effect:IsA("BlurEffect") or effect:IsA("ColorCorrectionEffect") then
                    if string.find(effect.Name:lower(), "flash") or string.find(effect.Name:lower(), "smoke") or string.find(effect.Name:lower(), "bang") then
                        effect.Enabled = false
                    end
                end
            end
        end
        
        -- Auto Kill (simplified - would need game-specific implementation)
        if Features.AutoKill then
            -- This would require game-specific remote event knowledge
            -- Placeholder for implementation
        end
    end)
end)
