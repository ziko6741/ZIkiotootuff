local function createMenuToggle(parent, labelText, defaultValue, callback)
	local row = Instance.new("Frame")
	row.Size = UDim2.new(1, 0, 0, 35)
	row.BackgroundTransparency = 1
	row.Parent = parent
	
	local textLabel = Instance.new("TextLabel")
	textLabel.Size = UDim2.new(0, 250, 1, 0)
	textLabel.BackgroundTransparency = 1
	textLabel.Text = labelText
	textLabel.Font = Enum.Font.GothamSemibold
	textLabel.TextColor3 = Color3.fromRGB(180, 200, 220)
	textLabel.TextSize = 12
	textLabel.TextXAlignment = Enum.TextXAlignment.Left
	textLabel.Parent = row
	
	local toggleBtn = Instance.new("TextButton")
	toggleBtn.Size = UDim2.new(0, 50, 0, 22)
	toggleBtn.Position = UDim2.new(1, -60, 0.5, -11)
	toggleBtn.BackgroundColor3 = defaultValue and Color3.fromRGB(0, 120, 255) or Color3.fromRGB(30, 35, 50)
	toggleBtn.Text = defaultValue and "ON" or "OFF"
	toggleBtn.Font = Enum.Font.GothamBold
	toggleBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
	toggleBtn.TextSize = 10
	toggleBtn.Parent = row
	Instance.new("UICorner", toggleBtn).CornerRadius = UDim.new(0, 4)
	
	local active = defaultValue
	toggleBtn.MouseButton1Click:Connect(function()
		active = not active
		toggleBtn.BackgroundColor3 = active and Color3.fromRGB(0, 120, 255) or Color3.fromRGB(30, 35, 50)
		toggleBtn.Text = active and "ON" or "OFF"
		callback(active)
	end)
end

local function createMenuSlider(parent, labelText, min, max, defaultValue, callback)
	local row = Instance.new("Frame")
	row.Size = UDim2.new(1, 0, 0, 45)
	row.BackgroundTransparency = 1
	row.Parent = parent
	
	local textLabel = Instance.new("TextLabel")
	textLabel.Size = UDim2.new(1, 0, 0, 20)
	textLabel.BackgroundTransparency = 1
	textLabel.Text = labelText .. ": " .. defaultValue
	textLabel.Font = Enum.Font.GothamSemibold
	textLabel.TextColor3 = Color3.fromRGB(180, 200, 220)
	textLabel.TextSize = 12
	textLabel.TextXAlignment = Enum.TextXAlignment.Left
	textLabel.Parent = row
	
	local sliderTrack = Instance.new("Frame")
	sliderTrack.Size = UDim2.new(1, -20, 0, 6)
	sliderTrack.Position = UDim2.new(0, 0, 0, 25)
	sliderTrack.BackgroundColor3 = Color3.fromRGB(25, 30, 45)
	sliderTrack.Parent = row
	Instance.new("UICorner", sliderTrack).CornerRadius = UDim.new(1, 0)
	
	local sliderFill = Instance.new("Frame")
	sliderFill.Size = UDim2.new((defaultValue - min) / (max - min), 0, 1, 0)
	sliderFill.BackgroundColor3 = Color3.fromRGB(0, 120, 255)
	sliderFill.BorderSizePixel = 0
	sliderFill.Parent = sliderTrack
	Instance.new("UICorner", sliderFill).CornerRadius = UDim.new(1, 0)
	
	local sliderBtn = Instance.new("TextButton")
	sliderBtn.Size = UDim2.new(0, 14, 0, 14)
	sliderBtn.Position = UDim2.new((defaultValue - min) / (max - min), -7, 0.5, -7)
	sliderBtn.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
	sliderBtn.Text = ""
	sliderBtn.Parent = sliderTrack
	Instance.new("UICorner", sliderBtn).CornerRadius = UDim.new(1, 0)
	
	local baseDragging = false
	local function parseInput(input)
		local relativeX = math.clamp(input.Position.X - sliderTrack.AbsolutePosition.X, 0, sliderTrack.AbsoluteSize.X)
		local ratio = relativeX / sliderTrack.AbsoluteSize.X
		sliderBtn.Position = UDim2.new(ratio, -7, 0.5, -7)
		sliderFill.Size = UDim2.new(ratio, 0, 1, 0)
		local finalVal = math.round(min + (ratio * (max - min)))
		textLabel.Text = labelText .. ": " .. finalVal
		callback(finalVal)
	end
	
	sliderBtn.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then baseDragging = true end
	end)
	UserInputService.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then baseDragging = false end
	end)
	UserInputService.InputChanged:Connect(function(input)
		if baseDragging and input.UserInputType == Enum.UserInputType.MouseMovement then parseInput(input) end
	end)
end

-- --- 6. TICK MECHANIC LOOPS & REGISTRATIONS ---

-- Character Model Tracker Loop
RunService.RenderStepped:Connect(function()
	local character = player.Character
	if character then
		local humanoid = character:FindFirstChildOfClass("Humanoid")
		if humanoid then
			-- Enforce Speed Matrix values locally
			if humanoid.WalkSpeed ~= MovementConfigs.WalkSpeed then
				humanoid.WalkSpeed = MovementConfigs.WalkSpeed
			end
			-- Enforce Jump Matrix values locally
			if humanoid.JumpPower ~= MovementConfigs.JumpPower then
				humanoid.UseJumpPower = true
				humanoid.JumpPower = MovementConfigs.JumpPower
			end
		end
		
		-- Noclip Character Physics Processor
		if MovementConfigs.FlightEnabled then
			for _, part in pairs(character:GetDescendants()) do
				if part:IsA("BasePart") then
					part.CanCollide = false
				end
			end
		end
	end
	
	-- Global Ambient Lighting Matrix Enforcement
	if VisualsConfigs.FullBrightActive then
		Lighting.Ambient = Color3.fromRGB(255, 255, 255)
		Lighting.OutdoorAmbient = Color3.fromRGB(255, 255, 255)
	end
	
	-- Interactive Neon Color Cycling Frame Border
	if InterfaceConfigs.RainbowTheme then
		local hue = (tick() % 4) / 4
		uiStroke.Color = Color3.fromHSV(hue, 0.8, 1)
	else
		uiStroke.Color = Color3.fromRGB(0, 120, 255)
	end
end)

-- Character Native Engine Highlighting Engine
local function refreshChamsVisuals()
	for _, targetPlayer in pairs(Players:GetPlayers()) do
		if targetPlayer ~= player and targetPlayer.Character then
			local char = targetPlayer.Character
			local existingHighlight = char:FindFirstChild("AdminHighlight")
			
			if VisualsConfigs.ChamsAlwaysOnTop then
				if not existingHighlight then
					existingHighlight = Instance.new("Highlight")
					existingHighlight.Name = "AdminHighlight"
					existingHighlight.Parent = char
				end
				existingHighlight.FillColor = VisualsConfigs.ChamsFillColor
				existingHighlight.OutlineColor = VisualsConfigs.ChamsOutlineColor
				existingHighlight.FillTransparency = VisualsConfigs.ChamsTransparency
				existingHighlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
			else
				if existingHighlight then
					existingHighlight:Destroy()
				end
			end
		end
	end
end

-- Hook highlight routine to runtime steps
RunService.Heartbeat:Connect(refreshChamsVisuals)

-- Infinite Jump Hook Connection
UserInputService.JumpRequest:Connect(function()
	if MovementConfigs.InfiniteJump and player.Character then
		local humanoid = player.Character:FindFirstChildOfClass("Humanoid")
		if humanoid then
			humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
		end
	end
end)

-- Weapon/Tool Trail Generation System
local function trackEquippedItemTrails(child)
	if child:IsA("Tool") and CombatConfigs.CustomWeaponTrail then
		local handle = child:FindFirstChild("Handle")
		if handle then
			local a0 = handle:FindFirstChild("TrailAtt0") or Instance.new("Attachment", handle)
			a0.Name = "TrailAtt0"
			a0.Position = Vector3.new(0, 1, 0)
			
			local a1 = handle:FindFirstChild("TrailAtt1") or Instance.new("Attachment", handle)
			a1.Name = "TrailAtt1"
			a1.Position = Vector3.new(0, -1, 0)
			
			local itemTrail = handle:FindFirstChild("AdminItemTrail") or Instance.new("Trail", handle)
			itemTrail.Name = "AdminItemTrail"
			itemTrail.Attachment0 = a0
			itemTrail.Attachment1 = a1
			itemTrail.Color = ColorSequence.new(Color3.fromRGB(0, 150, 255), Color3.fromRGB(255, 0, 100))
			itemTrail.Lifetime = 0.4
		end
	end
end

player.CharacterAdded:Connect(function(char)
	char.ChildAdded:Connect(trackEquippedItemTrails)
end)
if player.Character then
	player.Character.ChildAdded:Connect(trackEquippedItemTrails)
end
